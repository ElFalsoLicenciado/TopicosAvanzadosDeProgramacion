    private void btnPlayPauseActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_btnPlayPauseActionPerformed
        if( ! songLoaded) {
            JOptionPane.showMessageDialog(null,"No hay ninguna cancion en tu playlist", "Error", JOptionPane.ERROR_MESSAGE);
            return;
        }

        if( ! play)
        {
            btnPlayPause.setIcon(new javax.swing.ImageIcon(getClass().getResource("/images/pause.png")));
            player.playCurrentSong();
            play = !play;
        }
        else {
            btnPlayPause.setIcon(new javax.swing.ImageIcon(getClass().getResource("/images/play.png")));
            player.pauseSong();
            play = !play;
        }
    }//GEN-LAST:event_btnPlayPauseActionPerformed

    private void btnStopActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_btnStopActionPerformed
        if ( ! play){
            player.stopSong();
            setPlaybackSliderValue(0);
            return;
        }

        btnPlayPause.setIcon(new javax.swing.ImageIcon(getClass().getResource("/images/play.png")));
        setPlaybackSliderValue(0);
        player.stopSong();
        play = false;
    }//GEN-LAST:event_btnStopActionPerformed

    }

    public void savePlaylists(ArrayList<Playlist> l){
        playlists = l;
    }

    public boolean saveCurrentPlay(Playlist nuevaCurrent){
        for (Playlist p : playlists)
        {
            if (p.getidP().equals(nuevaCurrent.getidP())) {
                currentPlay = nuevaCurrent;
                System.out.println(nuevaCurrent.getidP());
                if( ! nuevaCurrent.getListado().isEmpty()) {
                    currentCancion = nuevaCurrent.getListado().getFirst();
                    songLoaded = true;
                    changelabelNombreCancion(currentCancion);
                    player.loadPlaylist(currentPlay);
                }
                else {
                    songLoaded = false;
                    labelNombreCancion.setText("No hay canciones en tu playlist");
                    setPlaybackSliderValue(0);
                    updatePlaybackSlider();

                }

                return true;
            }
        }
        JOptionPane.showMessageDialog(null, "La playlist no ha sido guardada","Error", JOptionPane.ERROR_MESSAGE);
        return false;
    }

    public void setPlaybackSliderValue(int value) {
        sliderTime.setValue(value);
    }
    public void updatePlaybackSlider() {
        sliderTime.setMaximum(0);

        Hashtable<Integer, JLabel> labelTable = new Hashtable<>();

        JLabel labelBeginning = new JLabel("--:--");
        labelBeginning.setFont(new Font("Dialog", Font.BOLD, 18));
        labelBeginning.setForeground(Color.BLACK);

        JLabel labelEnd =  new JLabel("--:--");
        labelEnd.setFont(new Font("Dialog", Font.BOLD, 18));
        labelEnd.setForeground(Color.BLACK);

        labelTable.put(0, labelBeginning);
        labelTable.put(1, labelEnd);

        sliderTime.setLabelTable(labelTable);
        sliderTime.setPaintLabels(true);
    }

    public void updatePlaybackSlider(Cancion song) {
        // update max count for slider
        sliderTime.setMaximum(song.getMp3File().getFrameCount());

        Hashtable<Integer, JLabel> labelTable = new Hashtable<>();

        // beginning will be 00:00
        JLabel labelBeginning = new JLabel("00:00");
        labelBeginning.setFont(new Font("Dialog", Font.BOLD, 18));
        labelBeginning.setForeground(Color.BLACK);

        // end will vary depending on the song
        JLabel labelEnd =  new JLabel(song.getSongLength());
        labelEnd.setFont(new Font("Dialog", Font.BOLD, 18));
        labelEnd.setForeground(Color.BLACK);

        labelTable.put(0, labelBeginning);
        labelTable.put(song.getMp3File().getFrameCount(), labelEnd);

        sliderTime.setLabelTable(labelTable);
        sliderTime.setPaintLabels(true);
    }
