    public PaginaPlaylist(PaginaReproductor reproductor,

            Cancion ejemplo = new Cancion("Instant Crush",
                    "Daft Punk",
                    "C:\\Users\\User\\Music\\Music For Deaf People\\278 - Daft Punk - Instant Crush (feat. Julian Casablancas).mp3");

            workingPlaylists.add(init);
            workingCurrent = init;

            savingPlaylist();
            reproductor.saveCurrentPlay(workingCurrent);
        }

        updatePlayCombo();
        updateItemList();

        setLocationRelativeTo(null);
        getContentPane().setBackground(Color.ORANGE);

    }

    private void btnAddActionPerformed(ActionEvent evt) { //Agregar archivo

        int response;
        JFileChooser fileSelector = new JFileChooser("C:/Users/User/Music/Music For Deaf People");

            workingCurrent.addSong(adicion);

        }
    }

    private void btnNewPlyActionPerformed(ActionEvent evt) {
        Playlist nueva = new Playlist("[Nueva]");
        workingPlaylists.add(nueva);
        updatePlayCombo();
        PlayCombo.setSelectedItem(nueva);
        workingCurrent = nueva;
        updateItemList();
    }

    private void btnRemPlayActionPerformed(ActionEvent evt) {
        if( (!workingPlaylists.isEmpty())) {

            Playlist remocion = (Playlist) PlayCombo.getSelectedItem();

            if(remocion != null)
            {
                PlayCombo.setSelectedIndex(0);
                workingPlaylists.remove(remocion);
                updatePlayCombo();
            }
        }
        else
            JOptionPane.showMessageDialog(null,
                    "No hay nada que borrar.",
                    "Error al borrar",
                    JOptionPane.ERROR_MESSAGE
            );
    }

    private void btnRenameActionPerformed(ActionEvent evt) {
        if(!workingPlaylists.isEmpty()) {
            int index = PlayCombo.getSelectedIndex();

            if(index != -1)
            {
                String nuevoNombre = JOptionPane.showInputDialog(null,
                        "Renombrar",
                        "Â¿Que nombre quieres ponerle?",
                        JOptionPane.QUESTION_MESSAGE
                );

                if ( nuevoNombre.isEmpty()) return;
                workingPlaylists.get(index).setNombrePlaylist(nuevoNombre);
                updatePlayCombo();
            }
        }
        else
            JOptionPane.showMessageDialog(null,
                    "No hay nada que editar.",
                    "Error al editar",
                    JOptionPane.ERROR_MESSAGE
            );

    }

    private void btnUpActionPerformed(ActionEvent evt) {
        int index = ItemList.getSelectedIndex();
        if(index != -1 && index < ItemList.getModel().getSize() -1 ) {
            Cancion bajar, subir;

            subir = ItemList.getModel().getElementAt(index);
            bajar = ItemList.getModel().getElementAt(index+1);

            workingCurrent.switchSong(subir, bajar);
            updateItemList();
        }
    }

    private void btnDownActionPerformed(ActionEvent evt) {
        int index = ItemList.getSelectedIndex();
        if(index > 1) {
            Cancion bajar, subir;

            subir = ItemList.getModel().getElementAt(index);
            bajar = ItemList.getModel().getElementAt(index-1);

            workingCurrent.switchSong(subir, bajar);
            updateItemList();
        }
    }

    public void updatePlayCombo(){
        modelPlayList.removeAllElements();
        for(Playlist lista: workingPlaylists)
        {
            modelPlayList.addElement(lista);
        }
        PlayCombo.setModel(modelPlayList);

    }

    public void updateItemList(){
        modelSongList.clear();
        for(Cancion rola: workingCurrent.getListado())
        {
            modelSongList.addElement(rola);
        }
        ItemList.setModel(modelSongList);
    }

    private void savingPlaylist() {
        originalPlaylists.clear();
        originalPlaylists.addAll(clonePlaylists(workingPlaylists));

        reproductor.savePlaylists(originalPlaylists);
    }

    private void setCurrentPlaylist() {
        reproductor.saveCurrentPlay((Playlist) PlayCombo.getSelectedItem());
    }

    }

    private ArrayList<Playlist> clonePlaylists(ArrayList<Playlist> src) {
        ArrayList<Playlist> copy = new ArrayList<>();
        for (Playlist p : src) {
            copy.add(p.clone());
        }
        return copy;
    }

    private Playlist findMatchingPlaylist(ArrayList<Playlist> list, Playlist original) {
        for (Playlist p : list) {
            if (p.getidP().equals(original.getidP())) {
                return p;
            }
        }
        return null;
    }
