    private void btnLoopActionPerformed(java.awt.event.ActionEvent evt) {
        if( ! loop)
        {
            btnLoop.setIcon(new javax.swing.ImageIcon(getClass().getResource("/images/loop.png")));
            /* Ejecutar archivo
            * Condicion para verificar archivos en la playlist
            */
            loop = !loop;
        }
        else {
            btnLoop.setIcon(new javax.swing.ImageIcon(getClass().getResource("/images/cycle.png")));
            /* Pausar archivo
            */
            loop = !loop;
        }
    }

    private void btnPlayPauseActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_btnPlayPauseActionPerformed
        if( ! songLoaded) {
            JOptionPane.showMessageDialog(null,"No hay ninguna cancion en tu playlist", "Error", JOptionPane.ERROR_MESSAGE);
            return;
        }

        if( ! play)
        {
            btnPlayPause.setIcon(new javax.swing.ImageIcon(getClass().getResource("/images/pause.png")));
            /* Ejecutar archivo
            * Condicion para verificar archivos en la playlist
            */
            play = !play;
        }
        else {
            btnPlayPause.setIcon(new javax.swing.ImageIcon(getClass().getResource("/images/play.png")));
            /* Pausar archivo
            */
            play = !play;
        }
    }//GEN-LAST:event_btnPlayPauseActionPerformed

    private void changelabelNombreCancion(Cancion cancion) {
        labelNombreCancion.setText(cancion.toString());
    }
    
    public void savePlaylists(ArrayList<Playlist> l){
        playlists = l;
    }
    
    public boolean saveCurrentPlay(Playlist nuevaCurrent){
        for (Playlist p : playlists)
        {
            if (p.getidP().equals(nuevaCurrent.getidP())) {
                currentPlay = nuevaCurrent;
                System.out.println(nuevaCurrent.getidP());
                if( ! nuevaCurrent.getListado().isEmpty()) {
                    currentCancion = nuevaCurrent.getListado().getFirst();
                    songLoaded = true;
                    changelabelNombreCancion(currentCancion);
                }
                else {
                    songLoaded = false;
                    labelNombreCancion.setText("No hay canciones en tu playlist");
                }

                return true;
            }
        }
        JOptionPane.showMessageDialog(null, "La playlist no ha sido guardada","Error", JOptionPane.ERROR_MESSAGE);
        return false;
    }
